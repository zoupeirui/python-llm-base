    "# --- 2. 自定义 PyTorch Dataset ---\n",
    "import torch\n",
    "from torch.utils.data import Dataset, DataLoader\n",
    "\n",
    "class VQADataset(Dataset):\n",
    "    \"\"\"多模态 VQA 微调数据集\"\"\"\n",
    "    def __init__(self, annotations, processor, max_len=128):\n",
    "        self.annotations = annotations\n",
    "        self.processor = processor\n",
    "        self.max_len = max_len\n",
    "\n",
    "    def __len__(self):\n",
    "        return len(self.annotations)\n",
    "\n",
    "    def __getitem__(self, idx):\n",
    "        item = self.annotations[idx]\n",
    "        image = Image.open(item[\"image\"]).convert(\"RGB\")\n",
    "        \n",
    "        # 构造 Qwen2.5-VL 标准的 Chat 格式消息\n",
    "        messages = [\n",
    "            {\n",
    "                \"role\": \"user\",\n",
    "                \"content\": [\n",
    "                    {\"type\": \"image\", \"image\": image}, # 图片占位\n",
    "                    {\"type\": \"text\", \"text\": item[\"conversations\"][0][\"content\"]} # 用户问题\n",
    "                ]\n",
    "            }\n",
    "        ]\n",
    "        \n",
    "        # 使用 Processor 的 chat template 将消息转为模型输入\n",
    "        text = self.processor.apply_chat_template(messages, tokenize=False, add_generation_prompt=True)\n",
    "        # 拼接上 assistant 的回答 (训练时需要让模型学会生成这段文字)\n",
    "        assistant_answer = item[\"conversations\"][1][\"content\"]\n",
    "        full_text = text + assistant_answer\n",
    "        \n",
    "        # Processor 同时处理图片和文本\n",
    "        inputs = self.processor(\n",
    "            text=[full_text], images=[image], \n",
    "            padding=\"max_length\", max_length=self.max_len, truncation=True,\n",
    "            return_tensors=\"pt\"\n",
    "        )\n",
    "        \n",
    "        # 去掉 batch 维度并构造 labels\n",
    "        input_ids = inputs[\"input_ids\"].squeeze(0)\n",
    "        attention_mask = inputs[\"attention_mask\"].squeeze(0)\n",
    "        pixel_values = inputs[\"pixel_values\"].squeeze(0) if \"pixel_values\" in inputs else None\n",
    "        image_grid_thw = inputs[\"image_grid_thw\"].squeeze(0) if \"image_grid_thw\" in inputs else None\n",
    "        \n",
    "        # 构造 labels: 将 prompt 部分的 token 设为 -100 (不计算 loss)，只在 assistant 回答部分计算 loss\n",
    "        labels = input_ids.clone()\n",
    "        # 找到 assistant 回答开始的位置（简化处理：对 prompt 部分全部掩码）\n",
    "        prompt_inputs = self.processor(\n",
    "            text=[text], images=[image],\n",
    "            padding=\"max_length\", max_length=self.max_len, truncation=True,\n",
    "            return_tensors=\"pt\"\n",
    "        )\n",
    "        prompt_len = (prompt_inputs[\"attention_mask\"].squeeze(0) == 1).sum().item()\n",
    "        labels[:prompt_len] = -100 # prompt 部分不参与 Loss 计算\n",
    "        \n",
    "        result = {\"input_ids\": input_ids, \"attention_mask\": attention_mask, \"labels\": labels}\n",

    "        if image_grid_thw is not None:\n",
    "            result[\"image_grid_thw\"] = image_grid_thw\n",
    "        return result\n",
    "\n",

    "## 步骤 5-6：训练循环与验证评估\n",
    "\n",
    "### ① 知识讲解\n",
    "由于大模型极其吃显存，单卡 Batch Size 通常只能设为 1。为了让训练稳定，我们使用**梯度累积 (Gradient Accumulation)**——连续做多个小 Batch 的前向传播，把梯度攒起来，每隔 N 步才统一更新一次参数。等效 Batch Size = 实际 Batch Size × 累积步数。\n",
    "\n",
    "### ② 为什么这么做\n",
    "Batch Size 太小（如 1），每一步的梯度噪声极大，模型像喝醉了酒一样到处乱撞。梯度累积让模型能在资源有限的条件下，依然享受大 Batch 的稳定性。"
   ]
  },
  {
   "cell_type": "code",
   "execution_count": null,
   "metadata": {},
   "outputs": [],
   "source": [
    "# ③ 完整代码\n",
    "optimizer = optim.AdamW(model.parameters(), lr=2e-4, weight_decay=0.01)\n",
    "\n",
    "epochs = 3\n",
    "gradient_accumulation_steps = 4 # 每 4 步累积一次梯度，等效 batch_size = 4\n",
    "log_interval = 10               # 每 10 步打印一次 Loss\n",
    "\n",
    "# 记录训练过程的 Loss，用于后续可视化\n",
    "train_loss_history = []\n",
    "val_loss_history = []\n",
    "\n",
    "for epoch in range(epochs):\n",
    "    # --- 训练阶段 ---\n",
    "    model.train()\n",
    "    epoch_loss = 0.0\n",
    "    optimizer.zero_grad() # 注意：梯度累积时，zero_grad 放在循环外面\n",
    "    \n",
    "    for step, batch in enumerate(train_loader):\n",
    "        # 将所有张量送入 GPU\n",
    "        batch = {k: v.to(device) for k, v in batch.items()}\n",
    "        \n",
    "        # 前向传播（HF 模型传入 labels 后会自动计算 CrossEntropy Loss）\n",
    "        outputs = model(**batch)\n",
    "        loss = outputs.loss\n",
    "        \n",
    "        # 将 Loss 除以累积步数，使得最终的等效梯度大小与大 Batch 一致\n",
    "        loss = loss / gradient_accumulation_steps\n",
    "        loss.backward()\n",
    "        \n",
    "        # 每累积 N 步后，统一更新一次参数\n",
    "        if (step + 1) % gradient_accumulation_steps == 0:\n",
    "            torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0) # 梯度裁剪防止爆炸\n",
    "            optimizer.step()\n",
    "            optimizer.zero_grad()\n",
    "        \n",
    "        epoch_loss += loss.item() * gradient_accumulation_steps\n",
    "        \n",
    "        if (step + 1) % log_interval == 0:\n",
    "            avg_loss = epoch_loss / (step + 1)\n",
    "            print(f\"  Epoch {epoch+1} Step {step+1}/{len(train_loader)} | Loss: {avg_loss:.4f}\")\n",
    "    \n",
    "    avg_train_loss = epoch_loss / len(train_loader)\n",
    "    train_loss_history.append(avg_train_loss)\n",
    "    \n",
    "    # --- 验证阶段 ---\n",
    "    model.eval()\n",
    "    val_loss = 0.0\n",
    "    with torch.no_grad():\n",
    "        for batch in val_loader:\n",
    "            batch = {k: v.to(device) for k, v in batch.items()}\n",
    "            outputs = model(**batch)\n",
    "            val_loss += outputs.loss.item()\n",
    "    avg_val_loss = val_loss / len(val_loader)\n",
    "    val_loss_history.append(avg_val_loss)\n",
    "    \n",
    "    print(f\"\\n{'='*60}\")\n",
    "    print(f\"Epoch {epoch+1}/{epochs} | Train Loss: {avg_train_loss:.4f} | Val Loss: {avg_val_loss:.4f}\")\n",
    "    print(f\"{'='*60}\\n\")"

    "### 💡 深度避坑与实践建议\n",
    "- **⑤ 容易踩坑**：梯度累积时，`loss = loss / gradient_accumulation_steps` 这一步极容易被遗漏！不除以累积步数的话，梯度会被放大 N 倍，训练立刻爆炸（Loss 变成 NaN）。另外，`torch.nn.utils.clip_grad_norm_` 是大模型训练的救命稻草——梯度爆炸时它会强制把梯度"削平"。\n",
